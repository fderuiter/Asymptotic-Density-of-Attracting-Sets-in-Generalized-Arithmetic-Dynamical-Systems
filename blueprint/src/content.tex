\chapter{Structural Prerequisites}

\begin{theorem}[Coprime Invertibility]
\lean{ArithmeticDynamics.coprime_implies_bijective_mod_d}
\leanok
If the multipliers are coprime to the modulus $d$, the map's action on the residue classes modulo $d$ is perfectly bijective.
\end{theorem}

\begin{theorem}
\lean{ArithmeticDynamics.dynamical_hensel_lift}
\leanok
If the aggregate multiplier of a cycle $(A) - 1$ is coprime to the modulus $d$, the cycle lifts to a unique invariant solution modulo $d^n$.
\end{theorem}

\begin{theorem}
\lean{ArithmeticDynamics.is_stochastic_matrix}
\leanok
The transition matrix of the ArithmeticDynamics map conserves probability mass.
\end{theorem}
